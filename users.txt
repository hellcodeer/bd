local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")
local Players = game:GetService("Players")

local LocalPlayer = Players.LocalPlayer
local PlayerGui = LocalPlayer:WaitForChild("PlayerGui")

local Library = {}
Library.__index = Library

local Theme = {
    Background = Color3.fromRGB(22, 22, 22),
    Topbar = Color3.fromRGB(28, 28, 28),
    Accent = Color3.fromRGB(0, 170, 255),
    TextActive = Color3.fromRGB(255, 255, 255),
    TextInactive = Color3.fromRGB(130, 130, 130),
    Button = Color3.fromRGB(35, 35, 35),
    Hover = Color3.fromRGB(48, 48, 48),
    Stroke = Color3.fromRGB(50, 50, 50),
    ToggleOff = Color3.fromRGB(45, 45, 45),
    ToggleOn = Color3.fromRGB(0, 200, 150),
    ToggleCircle = Color3.fromRGB(255, 255, 255)
}

function Library:CreateWindow(Name)

    if PlayerGui:FindFirstChild("mef") then
        PlayerGui.mef:Destroy()
    end

    local ScreenGui = Instance.new("ScreenGui")
    ScreenGui.Name = "mef"
    ScreenGui.ResetOnSpawn = false
    ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    ScreenGui.Parent = PlayerGui

    local Main = Instance.new("Frame")
    Main.Parent = ScreenGui
    Main.Size = UDim2.new(0, 650, 0, 420)
    Main.Position = UDim2.new(0.5, -325, 0.5, -210)
    Main.BackgroundColor3 = Theme.Background
    Main.BorderSizePixel = 0
    Main.ClipsDescendants = true

    Main.Size = UDim2.new(0, 0, 0, 0)
    TweenService:Create(
        Main,
        TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
        { Size = UDim2.new(0, 650, 0, 420) }
    ):Play()

    Instance.new("UICorner", Main).CornerRadius = UDim.new(0, 8)

    local Stroke = Instance.new("UIStroke")
    Stroke.Parent = Main
    Stroke.Color = Theme.Stroke
    Stroke.Thickness = 1

    local Topbar = Instance.new("Frame")
    Topbar.Parent = Main
    Topbar.Size = UDim2.new(1, 0, 0, 38)
    Topbar.BackgroundColor3 = Theme.Topbar
    Topbar.BorderSizePixel = 0

    local Title = Instance.new("TextLabel")
    Title.Parent = Topbar
    Title.BackgroundTransparency = 1
    Title.Position = UDim2.new(0, 14, 0, 0)
    Title.Size = UDim2.new(0, 140, 1, 0)
    Title.Font = Enum.Font.GothamBold
    Title.Text = Name or "mef"
    Title.TextColor3 = Theme.TextActive
    Title.TextSize = 16
    Title.TextXAlignment = Enum.TextXAlignment.Left

    local TabsHolder = Instance.new("ScrollingFrame")
    TabsHolder.Parent = Topbar
    TabsHolder.Position = UDim2.new(0, 150, 0, 0)
    TabsHolder.Size = UDim2.new(1, -160, 1, 0)
    TabsHolder.BackgroundTransparency = 1
    TabsHolder.ScrollBarThickness = 0
    TabsHolder.CanvasSize = UDim2.new(0, 0, 0, 0)

    local TabsLayout = Instance.new("UIListLayout")
    TabsLayout.Parent = TabsHolder
    TabsLayout.FillDirection = Enum.FillDirection.Horizontal
    TabsLayout.SortOrder = Enum.SortOrder.LayoutOrder
    TabsLayout.Padding = UDim.new(0, 12)

    TabsLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
        TabsHolder.CanvasSize = UDim2.new(0, TabsLayout.AbsoluteContentSize.X + 10, 0, 0)
    end)

    local Container = Instance.new("Frame")
    Container.Parent = Main
    Container.Position = UDim2.new(0, 12, 0, 50)
    Container.Size = UDim2.new(1, -24, 1, -62)
    Container.BackgroundTransparency = 1

    local Pages = {}
    local TabButtons = {}

    function Library:CreateTab(Text)

        local Button = Instance.new("TextButton")
        Button.Parent = TabsHolder
        Button.Size = UDim2.new(0, 0, 1, 0)
        Button.AutomaticSize = Enum.AutomaticSize.X
        Button.Text = Text
        Button.Font = Enum.Font.GothamMedium
        Button.TextSize = 14
        Button.TextColor3 = Theme.TextInactive
        Button.BackgroundTransparency = 1
        Button.BorderSizePixel = 0

        local AccentLine = Instance.new("Frame")
        AccentLine.Parent = Button
        AccentLine.AnchorPoint = Vector2.new(0.5, 1)
        AccentLine.Position = UDim2.new(0.5, 0, 1, 0)
        AccentLine.Size = UDim2.new(1, 0, 0, 2)
        AccentLine.BackgroundColor3 = Theme.TextActive
        AccentLine.BorderSizePixel = 0
        AccentLine.BackgroundTransparency = 1

        local Page = Instance.new("ScrollingFrame")
        Page.Parent = Container
        Page.Size = UDim2.new(1, 0, 1, 0)
        Page.Visible = false
        Page.CanvasSize = UDim2.new()
        Page.ScrollBarThickness = 3
        Page.ScrollBarImageColor3 = Theme.Stroke
        Page.BackgroundTransparency = 1

        local UIList = Instance.new("UIListLayout")
        UIList.Parent = Page
        UIList.Padding = UDim.new(0, 8)

        UIList:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            Page.CanvasSize = UDim2.new(0, 0, 0, UIList.AbsoluteContentSize.Y + 10)
        end)

        table.insert(Pages, Page)
        table.insert(TabButtons, {Button = Button, Line = AccentLine, Page = Page})

        local function ActivateTab()
            for _, item in ipairs(TabButtons) do
                item.Page.Visible = false
                
                TweenService:Create(item.Button, TweenInfo.new(0.2), {TextColor3 = Theme.TextInactive}):Play()
                TweenService:Create(item.Line, TweenInfo.new(0.2), {BackgroundTransparency = 1}):Play()
            end

            Page.Visible = true
            TweenService:Create(Button, TweenInfo.new(0.2), {TextColor3 = Theme.TextActive}):Play()
            TweenService:Create(AccentLine, TweenInfo.new(0.2), {BackgroundTransparency = 0}):Play()
        end

        Button.MouseButton1Click:Connect(ActivateTab)

        if #Pages == 1 then
            ActivateTab()
        end

        local Tab = {}
        Tab.Page = Page

        function Tab:AddLabel(Text)
            local Label = Instance.new("TextLabel")
            Label.Parent = Page
            Label.Size = UDim2.new(1, -8, 0, 28)
            Label.BackgroundTransparency = 1
            Label.Font = Enum.Font.Gotham
            Label.TextSize = 14
            Label.TextColor3 = Theme.TextActive
            Label.TextXAlignment = Enum.TextXAlignment.Left
            Label.Text = Text or "Label"

            return Label
        end

        function Tab:AddSeparator(Text)
            local Holder = Instance.new("Frame")
            Holder.Parent = Page
            Holder.Size = UDim2.new(1, -8, 0, 28)
            Holder.BackgroundTransparency = 1

            local Line1 = Instance.new("Frame")
            Line1.Parent = Holder
            Line1.AnchorPoint = Vector2.new(0, 0.5)
            Line1.Position = UDim2.new(0, 0, 0.5, 0)
            Line1.Size = UDim2.new(0.35, 0, 0, 1)
            Line1.BorderSizePixel = 0
            Line1.BackgroundColor3 = Theme.Stroke

            local Line2 = Instance.new("Frame")
            Line2.Parent = Holder
            Line2.AnchorPoint = Vector2.new(1, 0.5)
            Line2.Position = UDim2.new(1, 0, 0.5, 0)
            Line2.Size = UDim2.new(0.35, 0, 0, 1)
            Line2.BorderSizePixel = 0
            Line2.BackgroundColor3 = Theme.Stroke

            local Label = Instance.new("TextLabel")
            Label.Parent = Holder
            Label.AnchorPoint = Vector2.new(0.5, 0.5)
            Label.Position = UDim2.new(0.5, 0, 0.5, 0)
            Label.Size = UDim2.new(0.3, 0, 1, 0)
            Label.BackgroundTransparency = 1
            Label.Font = Enum.Font.GothamSemibold
            Label.TextSize = 13
            Label.TextColor3 = Theme.TextInactive
            Label.Text = Text or ""

            return Holder
        end

        function Tab:AddButton(Text, Callback)
            local Button = Instance.new("TextButton")
            Button.Parent = Page
            Button.Size = UDim2.new(1, -8, 0, 36)
            Button.BackgroundColor3 = Theme.Button
            Button.BorderSizePixel = 0
            Button.AutoButtonColor = false
            Button.Text = Text or "Button"
            Button.Font = Enum.Font.GothamMedium
            Button.TextSize = 14
            Button.TextColor3 = Theme.TextActive

            local Corner = Instance.new("UICorner")
            Corner.CornerRadius = UDim.new(0, 6)
            Corner.Parent = Button

            local Stroke = Instance.new("UIStroke")
            Stroke.Parent = Button
            Stroke.Color = Theme.Stroke
            Stroke.Thickness = 1

            Button.MouseEnter:Connect(function()
                TweenService:Create(Button, TweenInfo.new(0.15), {BackgroundColor3 = Theme.Hover}):Play()
            end)

            Button.MouseLeave:Connect(function()
                TweenService:Create(Button, TweenInfo.new(0.15), {BackgroundColor3 = Theme.Button}):Play()
            end)

            Button.MouseButton1Down:Connect(function()
                TweenService:Create(Button, TweenInfo.new(0.06), {Size = UDim2.new(1, -12, 0, 34)}):Play()
            end)

            Button.MouseButton1Up:Connect(function()
                TweenService:Create(Button, TweenInfo.new(0.06), {Size = UDim2.new(1, -8, 0, 36)}):Play()
                if Callback then task.spawn(Callback) end
            end)

            return Button
        end

        function Tab:AddToggle(Text, Default, Callback)
            local Toggled = Default or false

            local ToggleFrame = Instance.new("Frame")
            ToggleFrame.Parent = Page
            ToggleFrame.Size = UDim2.new(1, -8, 0, 36)
            ToggleFrame.BackgroundColor3 = Theme.Button
            ToggleFrame.BorderSizePixel = 0

            local Corner = Instance.new("UICorner")
            Corner.CornerRadius = UDim.new(0, 6)
            Corner.Parent = ToggleFrame

            local Stroke = Instance.new("UIStroke")
            Stroke.Parent = ToggleFrame
            Stroke.Color = Theme.Stroke
            Stroke.Thickness = 1

            local Label = Instance.new("TextLabel")
            Label.Parent = ToggleFrame
            Label.BackgroundTransparency = 1
            Label.Position = UDim2.new(0, 12, 0, 0)
            Label.Size = UDim2.new(1, -70, 1, 0)
            Label.Font = Enum.Font.GothamMedium
            Label.Text = Text or "Toggle"
            Label.TextColor3 = Theme.TextActive
            Label.TextSize = 14
            Label.TextXAlignment = Enum.TextXAlignment.Left

            local SwitchTrack = Instance.new("Frame")
            SwitchTrack.Parent = ToggleFrame
            SwitchTrack.AnchorPoint = Vector2.new(1, 0.5)
            SwitchTrack.Position = UDim2.new(1, -10, 0.5, 0)
            SwitchTrack.Size = UDim2.new(0, 42, 0, 22)
            SwitchTrack.BackgroundColor3 = Toggled and Theme.ToggleOn or Theme.ToggleOff
            SwitchTrack.BorderSizePixel = 0

            local TrackCorner = Instance.new("UICorner")
            TrackCorner.CornerRadius = UDim.new(1, 0)
            TrackCorner.Parent = SwitchTrack

            local SwitchCircle = Instance.new("Frame")
            SwitchCircle.Parent = SwitchTrack
            SwitchCircle.AnchorPoint = Vector2.new(0, 0.5)
            SwitchCircle.Position = Toggled and UDim2.new(1, -19, 0.5, 0) or UDim2.new(0, 3, 0.5, 0)
            SwitchCircle.Size = UDim2.new(0, 16, 0, 16)
            SwitchCircle.BackgroundColor3 = Theme.ToggleCircle
            SwitchCircle.BorderSizePixel = 0

            local CircleCorner = Instance.new("UICorner")
            CircleCorner.CornerRadius = UDim.new(1, 0)
            CircleCorner.Parent = SwitchCircle

            local ClickButton = Instance.new("TextButton")
            ClickButton.Parent = ToggleFrame
            ClickButton.Size = UDim2.new(1, 0, 1, 0)
            ClickButton.BackgroundTransparency = 1
            ClickButton.Text = ""

            local function UpdateState()
                local TargetColor = Toggled and Theme.ToggleOn or Theme.ToggleOff
                local TargetPos = Toggled and UDim2.new(1, -19, 0.5, 0) or UDim2.new(0, 3, 0.5, 0)

                TweenService:Create(SwitchTrack, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
                    BackgroundColor3 = TargetColor
                }):Play()

                TweenService:Create(SwitchCircle, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
                    Position = TargetPos
                }):Play()

                if Callback then
                    task.spawn(Callback, Toggled)
                end
            end

            ClickButton.MouseEnter:Connect(function()
                TweenService:Create(ToggleFrame, TweenInfo.new(0.15), {BackgroundColor3 = Theme.Hover}):Play()
            end)

            ClickButton.MouseLeave:Connect(function()
                TweenService:Create(ToggleFrame, TweenInfo.new(0.15), {BackgroundColor3 = Theme.Button}):Play()
            end)

            ClickButton.MouseButton1Click:Connect(function()
                Toggled = not Toggled
                UpdateState()
            end)

            return ToggleFrame
        end

        return Tab
    end

    local Dragging, DragStart, StartPos

    Topbar.InputBegan:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseButton1 then
            Dragging = true
            DragStart = Input.Position
            StartPos = Main.Position
        end
    end)

    UserInputService.InputEnded:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseButton1 then
            Dragging = false
        end
    end)

    UserInputService.InputChanged:Connect(function(Input)
        if Dragging and Input.UserInputType == Enum.UserInputType.MouseMovement then
            local Delta = Input.Position - DragStart
            TweenService:Create(
                Main,
                TweenInfo.new(0.05),
                {
                    Position = UDim2.new(
                        StartPos.X.Scale,
                        StartPos.X.Offset + Delta.X,
                        StartPos.Y.Scale,
                        StartPos.Y.Offset + Delta.Y
                    )
                }
            ):Play()
        end
    end)

    return Library
end

return Library
